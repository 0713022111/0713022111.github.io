```shell
SELECT
    BIRTHDAY,DATE_FORMAT(BIRTHDAY, '%m%d') - DATE_FORMAT(NOW(), '%m%d'),m.*
FROM
    member m
ORDER BY

IF (
    BIRTHDAY IS NULL
    OR BIRTHDAY = '',
    2,

IF (
    DATE_FORMAT(BIRTHDAY, '%m.%d') >= DATE_FORMAT(NOW(), '%m.%d'),
    0,
    1
)
) ASC,
 (
    DATE_FORMAT(BIRTHDAY, '%m%d') - DATE_FORMAT(NOW(), '%m%d')
) ASC
```
